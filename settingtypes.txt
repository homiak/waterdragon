#    If true, Water Dragons will spawn in mountains with Nests
water_nest_spawning (Nest Spawning) bool true

#    Water Dragon Nest Spawn Rate. 1 in x chance per chunk generated for a Water Dragon Nest to spawn. This setting only applies when Nest Spawning is enabled.
water_nest_spawn_rate (Nest Spawn Rate) int 128

#    If true, Water Dragons will spawn in roosts and caverns like Minecrafts Ice and Fire mod.
water_cavern_spawning (Rare Water and Pure Water Cavern Spawning) bool false

#    Water Dragon Cavern Spawn Rate. This setting only applies when Rare Water and Pure Water Cavern Spawning is enabled.
water_cavern_spawn_rate (Rare/Pure Water Cavern Spawn Rate) int 32

#    If true, Water Dragon breath will make nodes wet and Water Dragons will attempt to break blocks when stuck.
water_dragon_terrain_destruction (Terrain Destruction) bool true

#    Disable if stairs mod gives a crash (Exile and other games are known to cause this)
water_register_stairs (Register Stairs) bool true

#    Chance (by percentage of 100) for a special colour to occur on one of Water Dragons potential colour regions
water_unique_color_chance (Unique Colour Chance) int 65
